3@file ImageToAngles.py
4@brief Convert a black-on-white PNG drawing to SCARA waypoints.
5
6Every foreground (“ink”) pixel in the skeletonized PNG becomes one waypoint,

36#: SCARA link-lengths (inches)

45# ────────────────── 1. PNG → skeleton → graph-walk points ───────────────
46def parse_png(path: str | Path) -> List[Tuple[float, float, int]]:
47 """
48 @brief Skeletonize the PNG and traverse its pixel graph.
49
50 @param path Path to a *black-on-white* PNG image.
51 @return Ordered list of **(x, y, z)** waypoints in **pixel units**
52 where *z = 1* designates a pen lift between disconnected
53 strokes.
54
55 @throw FileNotFoundError If *path* cannot be opened.
56 """
57 # -- load & binarize --------------------------------------------------
58 img = cv2.imread(str(path), cv2.IMREAD_GRAYSCALE)
59 if img is None:
60 raise FileNotFoundError(f"Could not open image: {path}")
61
62 _, bw = cv2.threshold(
63 img, 0, 255, cv2.THRESH_BINARY_INV + cv2.THRESH_OTSU
64 ) # foreground → 255, background → 0
65
66 bw_bool = bw > 0
67 skel_bool = skeletonize(bw_bool) # 1-pixel-wide skeleton
68
69 # -- build adjacency graph -------------------------------------------
70 h, w = img.shape
71 ys, xs = np.nonzero(skel_bool)
72 pixels = set(zip(xs.tolist(), ys.tolist()))
73 if not pixels:
74 return []
75
76 nbrs: dict[Tuple[int, int], list[Tuple[int, int]]] = {p: [] for p in pixels}
77 for x, y in pixels:
78 for dx in (-1, 0, 1):
79 for dy in (-1, 0, 1):
80 if dx == 0 and dy == 0:
81 continue
82 nb = (x + dx, y + dy)
83 if nb in pixels:
84 nbrs[(x, y)].append(nb)
85
86 # -- pick a starting pixel (degree 1 endpoint if available) ----------
87 endpoints = [p for p, ngh in nbrs.items() if len(ngh) == 1]
88 start = endpoints[0] if endpoints else next(iter(pixels))
89
90 # -- depth-first traversal (visit each pixel exactly once) -----------
91 visited = {start}
92 stack = [start]
93 path: list[Tuple[int, int]] = []
94
95 while stack:
96 p = stack.pop()
97 path.append(p)
98 for q in nbrs[p]:
99 if q not in visited:
100 visited.add(q)
101 stack.append(q)
102
103 # -- convert pixel coords → (x, y, z) waypoints ---------------------
104 pts: List[Tuple[float, float, int]] = []
105 prev = path[0]
106 pts.append((float(prev[0]), float(h - 1 - prev[1]), 0)) # first point
107
108 for x, y in path[1:]:
109 if (x, y) not in nbrs[prev]:
110 # discontinuity → pen up then down
111 pts.append((float(prev[0]), float(h - 1 - prev[1]), 1))
112 pts.append((float(x), float(h - 1 - y), 0))
113 else:
114 pts.append((float(x), float(h - 1 - y), 0))
115 prev = (x, y)
116
117 return pts

120# ───────────────────── 2. Normalize to workspace ────────────────────────
121def normalize(pts: Sequence[Tuple[float, float, int]]) -> List[Tuple[float, float, int]]:
122 """
123 @brief Scale and center pixel waypoints into the SCARA workspace.
124
125 The drawing is **uniformly** scaled to fit inside *X_RANGE × Y_RANGE*
126 while preserving aspect ratio, and then translated so its centroid lies
127 at the center of that rectangle.
128
129 @param pts Waypoints from :pyfunc:`parse_png` in pixel units.
130 @return Same list of points, but in **inch units** relative to the
131 robot base.
132 """
133 xs = [p[0] for p in pts]
134 ys = [p[1] for p in pts]
135
136 sx = (X_RANGE[1] - X_RANGE[0]) / (max(xs) - min(xs))
137 sy = (Y_RANGE[1] - Y_RANGE[0]) / (max(ys) - min(ys))
138 s = min(sx, sy) # uniform scale
139
140 cx, cy = (max(xs) + min(xs)) / 2, (max(ys) + min(ys)) / 2
141 tx, ty = (X_RANGE[0] + X_RANGE[1]) / 2, (Y_RANGE[0] + Y_RANGE[1]) / 2
142
143 out: List[Tuple[float, float, int]] = []
144 for x, y, z in pts:
145 out.append(((x - cx) * s + tx, (y - cy) * s + ty, z))
146 return out

149# ──────────────────── 3. Distance-based down-sampling ───────────────────
150def downsample_by_dist(
151 pts: Sequence[Tuple[float, float, int]], min_dist: float
152) -> List[Tuple[float, float, int]]:
153 """
154 @brief Remove redundant waypoints that are closer than *min_dist*.
155
156 @param pts Input waypoints (inch units).
157 @param min_dist Minimum allowable spacing (inches).
158 Points with *z = 1* (pen-up) are **never** removed.
159 @return Filtered list with fewer than or equal to *len(pts)*
160 elements.
161 """
162 if min_dist <= 0 or not pts:
163 return list(pts)
164
165 out = [pts[0]]
166 last_x, last_y, _ = pts[0]
167
168 for x, y, z in pts[1:]:
169 dx, dy = x - last_x, y - last_y
170 if math.hypot(dx, dy) >= min_dist or z == 1:
171 out.append((x, y, z))
172 last_x, last_y = x, y
173 return out

176# ────────────────────── 4. Inverse kinematics ───────────────────────────
177def ik_xy(x: float, y: float) -> Tuple[float, float]:
178 """
179 @brief Two-link planar inverse kinematics.
180
181 @param x X-coordinate in inches.
182 @param y Y-coordinate in inches.
183 @return (θ₁, θ₂) in **degrees**. Both angles are made negative so that
184 positive robot motion matches the SCARA’s CCW convention.
185
186 @throw ValueError If *(x, y)* lies outside the reachable workspace.
187 """
188 r2 = x * x + y * y
189 c = (r2 - L1 * L1 - L2 * L2) / (2 * L1 * L2)
190 if abs(c) > 1:
191 raise ValueError(f"Point ({x:.2f}, {y:.2f}) out of reach")
192
193 s = math.sqrt(1 - c * c)
194 t2 = math.atan2(s, c) - math.radians(OFFSET_T2_DEG)
195
196 k1, k2 = L1 + L2 * c, L2 * s
197 t1 = math.atan2(y, x) - math.atan2(k2, k1)
198
199 return -abs(math.degrees(t1)), -abs(math.degrees(t2))
200
201
202def ik_path(pts: Sequence[Tuple[float, float, int]]) -> List[Tuple[float, float]]:
203 """
204 @brief Vectorized helper: apply :pyfunc:`ik_xy` to an entire path.
205
206 @param pts Waypoints **after** normalization/down-sampling.
207 @return List of *(θ₁, θ₂)* pairs in degrees.
208 """
209 return [ik_xy(x, y) for x, y, _ in pts]
